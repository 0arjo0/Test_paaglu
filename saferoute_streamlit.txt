# ─────────────────────────────────────────────────────────────────
def calc_score(row: dict) -> int:
    """Weighted safety score from stop attributes (0–100)."""
    lighting  = float(row.get("lighting_score",  row.get("lighting",  50)) or 50)
    footfall  = float(row.get("footfall_score",  row.get("footfall",  50)) or 50)
    cctv      = float(row.get("cctv_score",      row.get("cctv",      50)) or 50)
    incidents = float(row.get("incident_count",  row.get("incidents",  0)) or  0)
    isolation = float(row.get("isolation_score", row.get("isolation",  50)) or 50)
    emergency = float(row.get("emergency_access",row.get("emergency",  50)) or 50)
    inc_score = max(0.0, 100.0 - incidents * 2.0)
    iso_score = 100.0 - isolation
    return round(
        lighting  * 0.20 +
        footfall  * 0.18 +
        cctv      * 0.22 +
        inc_score * 0.20 +
        iso_score * 0.12 +
        emergency * 0.08
    )

def tier(score: int) -> str:
    if score >= 70: return "safe"
    if score >= 45: return "caution"
    return "danger"

def tier_color(t: str) -> str:
    return {"safe": "#69f0ae", "caution": "#ffab40", "danger": "#ff5252"}.get(t, "#7aadca")

def tier_badge(score: int) -> str:
    t = tier(score)
    cls = f"badge-{t}"
    label = t.capitalize()
    return f'<span class="badge {cls}">{label}</span>'

    aliases = {
        "stop_id":        ["stop_id","id","stopid"],
        "stop_name":      ["stop_name","name","stopname","station"],
        "latitude":       ["latitude","lat"],
        "longitude":      ["longitude","lng","lon"],
        "zone":           ["zone","area","district"],
        "lighting_score": ["lighting_score","lighting","lux","light"],
        "footfall_score": ["footfall_score","footfall","crowd","pedestrian"],
        "cctv_score":     ["cctv_score","cctv","camera","cameras"],
        "incident_count": ["incident_count","incidents","crimes","reports"],
        "isolation_score":["isolation_score","isolation","isolated"],
        "emergency_access":["emergency_access","emergency","ems"],
    }
    for target, srcs in aliases.items():
        if target not in df.columns:
            for s in srcs:
                if s in df.columns:
                    df[target] = df[s]
                    break
        if target not in df.columns:
            df[target] = 50 if "score" in target or target in ["lighting_score","footfall_score","cctv_score","isolation_score","emergency_access"] else (0 if target=="incident_count" else "Unknown")
    df["score"] = df.apply(lambda r: calc_score(r.to_dict()), axis=1)
    df["tier"]  = df["score"].apply(tier)
    return df

def _parse_routes(df: pd.DataFrame) -> pd.DataFrame:
    df = _normalize_cols(df)
    for col, default in [("route_id","R?"),("route_name","Route"),("route_type","Bus"),
                         ("stops_sequence",""),("base_time_min",30),("night_penalty",1.2),("operates_night",True)]:
        for alias in [col, col.replace("_",""), col.split("_")[0]]:
            if alias in df.columns:
                if col not in df.columns: df[col] = df[alias]
                break
        if col not in df.columns: df[col] = default
    return df

def _parse_incidents(df: pd.DataFrame) -> pd.DataFrame:
    df = _normalize_cols(df)
    for col, default in [("incident_id",""),("stop_id",""),("incident_type","Unknown"),
                         ("time_reported",""),("severity","Low"),("description","")]:
        if col not in df.columns:
            for alias in [col, col.replace("_",""), col.split("_")[0]]:
                if alias in df.columns: df[col] = df[alias]; break
            else: df[col] = default
    return df

def _dataframes_to_text(stops_df, routes_df, incidents_df) -> str:
    lines = ["=== TRANSIT STOPS ==="]
    for _, r in stops_df.iterrows():
        lines.append(f"Stop {r.get('stop_id','?')} '{r.get('stop_name','?')}' "
                     f"zone={r.get('zone','?')} "
                     f"safety_score={r.get('score', calc_score(r.to_dict()))} "
                     f"lighting={r.get('lighting_score','?')} "
                     f"footfall={r.get('footfall_score','?')} "
                     f"cctv={r.get('cctv_score','?')} "
                     f"incidents={r.get('incident_count','?')} "
                     f"tier={r.get('tier', tier(int(r.get('score',50))))}")
    lines.append("\n=== ROUTES ===")
    for _, r in routes_df.iterrows():
        lines.append(f"Route {r.get('route_id','?')} '{r.get('route_name','?')}' "
                     f"type={r.get('route_type','?')} "
                     f"stops={r.get('stops_sequence','?')} "
                     f"time={r.get('base_time_min','?')}min "
                     f"operates_night={r.get('operates_night','?')}")
    lines.append("\n=== INCIDENTS ===")
    for _, r in incidents_df.iterrows():
        lines.append(f"Incident at stop {r.get('stop_id','?')}: "
                     f"type={r.get('incident_type','?')} "
                     f"severity={r.get('severity','?')} "
                     f"time={r.get('time_reported','?')} "
                     f"desc={r.get('description','?')}")
    return "\n".join(lines)

def process_uploaded_files(files: list):
    """
    Accept multiple uploaded files (CSV, XLSX, PDF, TXT).
    Auto-detect type from column names or filename hints.
    """
    stops_frames, routes_frames, incident_frames, raw_texts = [], [], [], []

    for f in files:
        name = f.name.lower()
        try:
            # ── PDF ──────────────────────────────────────────────
            if name.endswith(".pdf"):
                with tempfile.NamedTemporaryFile(delete=False, suffix=".pdf") as tmp:
                    tmp.write(f.read())
                    tmp_path = tmp.name
                raw_texts.append(extract_text(tmp_path))
                os.unlink(tmp_path)

            # ── XLSX / XLS ───────────────────────────────────────
            elif name.endswith((".xlsx", ".xls")):
                xf = pd.ExcelFile(f)
                for sheet in xf.sheet_names:
                    df = xf.parse(sheet)
                    _classify_and_store(df, sheet, stops_frames, routes_frames, incident_frames)

            # ── CSV / TXT ────────────────────────────────────────
            elif name.endswith((".csv", ".txt")):
                try:
                    df = pd.read_csv(f)
                except Exception:
                    f.seek(0)
                    raw_texts.append(f.read().decode("utf-8", errors="replace"))
                    continue
                _classify_and_store(df, name, stops_frames, routes_frames, incident_frames)

        except Exception as e:
            st.warning(f"⚠ Could not parse {f.name}: {e}")

    # Merge & store
    if stops_frames:
        merged = pd.concat(stops_frames, ignore_index=True).drop_duplicates()
        st.session_state.stops_df = _parse_stops(merged)
    if routes_frames:
        merged = pd.concat(routes_frames, ignore_index=True).drop_duplicates()
        st.session_state.routes_df = _parse_routes(merged)
    if incident_frames:
        merged = pd.concat(incident_frames, ignore_index=True).drop_duplicates()
        st.session_state.incidents_df = _parse_incidents(merged)

    # Build full RAG text
    combined_text = "\n\n".join(raw_texts)
    if st.session_state.stops_df is not None:
        combined_text += "\n\n" + _dataframes_to_text(
            st.session_state.stops_df,
            st.session_state.routes_df if st.session_state.routes_df is not None else pd.DataFrame(),
            st.session_state.incidents_df if "incidents_df" in st.session_state and st.session_state.incidents_df is not None else pd.DataFrame(),
        )
    if combined_text.strip():
        _build_rag_from_text(combined_text)
    elif st.session_state.stops_df is None:
        st.error("No usable data found. Please check your file format and column names.")


def _classify_and_store(df: pd.DataFrame, hint: str, stops_frames, routes_frames, incident_frames):
    cols = set(c.lower().strip().replace(" ","_") for c in df.columns)
    hint_l = hint.lower()
    # Route hints
    if any(x in hint_l for x in ["route","transit_routes"]) or \
       any(x in cols for x in ["route_id","stops_sequence","route_name","night_penalty"]):
        routes_frames.append(df)
    # Incident hints
    elif any(x in hint_l for x in ["incident","crime","report"]) or \
         any(x in cols for x in ["incident_id","incident_type","severity"]):
        incident_frames.append(df)
    # Default → stops
    else:
        stops_frames.append(df)


def _build_rag_from_text(text: str):
    """Chunk → embed → vectorstore → RAG chain."""
    with st.spinner("🔍 Building RAG index…"):
        splitter = RecursiveCharacterTextSplitter(chunk_size=800, chunk_overlap=200)
        chunks   = splitter.split_text(text)
        try:
            embeddings = make_embeddings()
            vectordb   = Chroma.from_texts(chunks, embeddings, persist_directory=CHROMA_DIR)
            vectordb.persist()
            retriever  = vectordb.as_retriever(search_kwargs={"k": 4})
            llm        = make_llm()
            chain      = RetrievalQA.from_chain_type(
                llm=llm, retriever=retriever, return_source_documents=True
            )
            st.session_state.rag_chain = chain
            st.session_state.rag_ready = True
        except Exception as e:
            # If API is unreachable, mark rag_ready anyway for UI; chat will use fallback
            st.warning(f"⚠ RAG API unavailable ({e}). Running in offline demo mode.")
            st.session_state.rag_chain = None
            st.session_state.rag_ready = True   # data is loaded, just no live LLM

def compute_routes(origin_id: str, dest_id: str, safety_weight: float = 0.7) -> list:
    """
    Find all routes that contain both origin and dest (in order).
    Score = safety_weight * avg_safety + (1-safety_weight) * speed_score
    Returns list of dicts sorted best → worst.
    """
    stops_df   = st.session_state.stops_df
    routes_df  = st.session_state.routes_df
    incidents  = st.session_state.get("incidents_df", pd.DataFrame())

    if stops_df is None or routes_df is None:
        return []

    stop_lookup = {r["stop_id"]: r for _, r in stops_df.iterrows()}
    results = []

    for _, route in routes_df.iterrows():
        seq_raw = str(route.get("stops_sequence",""))
        seq = [s.strip() for s in re.split(r"[-,;|→ ]+", seq_raw) if s.strip()]
        try:
            oi = seq.index(origin_id)
            di = seq.index(dest_id)
        except ValueError:
            continue
        if oi >= di:
            continue

        segment = seq[oi : di + 1]
        seg_stops = [stop_lookup[s] for s in segment if s in stop_lookup]
        if not seg_stops:
            continue

        scores = [int(s.get("score", calc_score(dict(s)))) for s in seg_stops]
        avg_safety = sum(scores) / len(scores)
        min_safety = min(scores)

        # Incident penalty
        if not incidents.empty:
            inc_on_route = incidents[incidents["stop_id"].isin(segment)]
            penalty = inc_on_route["severity"].map({"High":15,"Medium":8,"Low":3}).sum()
        else:
            penalty = 0

        effective_safety = max(0, avg_safety - penalty)
        base_time = float(route.get("base_time_min", 30))
        night_pen = float(route.get("night_penalty",  1.2))
        night_time = base_time * night_pen

        # Speed score: inverse of time (scaled 0–100, assuming max 90 min)
        speed_score = max(0, 100 - night_time * (100/90))
        final = safety_weight * effective_safety + (1 - safety_weight) * speed_score

        results.append({
            "route":           route,
            "segment":         segment,
            "seg_stops":       seg_stops,
            "scores":          scores,
            "avg_safety":      round(effective_safety),
            "min_safety":      round(min_safety),
            "night_time":      round(night_time),
            "base_time":       round(base_time),
            "final_score":     round(final),
            "tier":            tier(round(effective_safety)),
            "route_type":      route.get("route_type","Bus"),
            "route_name":      route.get("route_name","Route"),
            "inc_on_route":    incidents[incidents["stop_id"].isin(segment)] if not incidents.empty else pd.DataFrame(),
        })

    results.sort(key=lambda x: x["final_score"], reverse=True)
    return results

            location=[lat_, lng_],
            radius=7,
            color=clr,
            fill=True,
            fill_color=clr,
            fill_opacity=0.75,
            weight=2,
            popup=folium.Popup(popup_html, max_width=220),
            tooltip=f"{name_} ({score_})"
        ).add_to(m)

    # ── Draw route polylines ────────────────────────────────────
    stop_latlng = {
        row["stop_id"]: (float(row["latitude"]), float(row["longitude"]))
        for _, row in stops_df.iterrows()
        if "stop_id" in row and "latitude" in row
    }

    for i, rc in enumerate(routes):
        coords = [stop_latlng[s] for s in rc["segment"] if s in stop_latlng]
        if len(coords) < 2:
            continue
        color   = ROUTE_PALETTE[i % len(ROUTE_PALETTE)]

        if dash:
            line_kw["dash_array"] = dash
        folium.PolyLine(**line_kw).add_to(m)

        # Start / end markers for selected route
        if i == selected_idx and coords:
            folium.Marker(
                coords[0],
                icon=folium.DivIcon(html=f'<div style="background:{color};color:#000;font-weight:700;padding:3px 7px;border-radius:6px;font-size:11px;white-space:nowrap;">FROM</div>'),
            ).add_to(m)

                coords[-1],
                icon=folium.DivIcon(html=f'<div style="background:{color};color:#000;font-weight:700;padding:3px 7px;border-radius:6px;font-size:11px;white-space:nowrap;">TO</div>'),
            ).add_to(m)

    return m._repr_html_()

            return result.get("result", "")

    # Offline fallback

def ask_chat(question: str) -> str:
    try:
        llm = make_llm()

        class AgentState(TypedDict):
            messages: List[Union[HumanMessage, AIMessage, SystemMessage]]

        def node(state: AgentState) -> AgentState:
            resp = llm.invoke(state["messages"])
            state["messages"].append(AIMessage(content=resp.content))
            return state

        g = StateGraph(AgentState)
        g.add_node("node1", node)
        g.add_edge(START, "node1")
        g.add_edge("node1", END)
        agent  = g.compile()
        result = agent.invoke({"messages": st.session_state.messages.copy()})
        return result["messages"][-1].content
    except Exception:
        return _offline_answer(question)

def _offline_answer(question: str) -> str:
    """Deterministic fallback when no API is reachable."""
    ql = question.lower()
    stops_df = st.session_state.stops_df
    if stops_df is None:
        return "No data loaded yet. Please upload files or load the sample dataset first."

    if any(w in ql for w in ["safest","safe stop","top stop"]):
        top = stops_df.nlargest(3,"score")[["stop_name","score"]].values.tolist()
        txt = ", ".join(f"**{n}** ({s})" for n,s in top)
        return f"The safest stops in the loaded data are: {txt}."

    if any(w in ql for w in ["dangerous","risk","worst","lowest"]):
        bot = stops_df.nsmallest(3,"score")[["stop_name","score"]].values.tolist()
        txt = ", ".join(f"**{n}** ({s})" for n,s in bot)
        return f"The highest-risk stops are: {txt}. These have poor CCTV, lighting, or high incident counts."

    if any(w in ql for w in ["score","calculat","formula","weight"]):
        return ("Safety score = **0.22×CCTV + 0.20×Lighting + 0.18×Footfall + "
                "0.20×(100−incidents×2) + 0.12×(100−isolation) + 0.08×Emergency**. "
                "Scores ≥70 = Safe · 45–69 = Caution · <45 = High Risk.")

    if "incident" in ql:
        inc = st.session_state.get("incidents_df")
        if inc is not None and not inc.empty:
            hi = inc[inc["severity"]=="High"]
            return (f"There are **{len(inc)} incidents** loaded. "
                    f"**{len(hi)} high-severity** incidents. "
                    f"Stops affected: {', '.join(inc['stop_id'].unique()[:5])}.")

    avg = int(stops_df["score"].mean())
    danger_cnt = int((stops_df["tier"]=="danger").sum())
    return (f"Network overview: **{len(stops_df)} stops** loaded. "
            f"Average safety score: **{avg}**. "
            f"**{danger_cnt}** stops rated high-risk. "
            f"Use the Trip Planner tab to explore route options.")
